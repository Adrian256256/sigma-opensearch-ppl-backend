source=windows-system-* | where ((EventID=7045 AND (LIKE(ServiceFileName, "%\\\\%\\%") OR LIKE(ServiceFileName, "%ADMIN$%"))) OR (LIKE(ParentImage, "%\\services.exe") AND (LIKE(Image, "%\\cmd.exe%") OR LIKE(Image, "%\\powershell.exe%") OR LIKE(Image, "%\\wmic.exe%")))) | stats dc(EventID) as unique_rules by span(@timestamp, 2m), ComputerName | where unique_rules >= 2
